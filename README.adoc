import java.awt.Color;
import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
import javax.imageio.ImageIO;

public class ImageToAscii {
// Characters from lightest (space) to darkest (@)
private static final String ASCII_CHARS = " .:-=+*#%@";

    public static String convertToAscii(BufferedImage image) {
        StringBuilder ascii = new StringBuilder();
        int width = image.getWidth();
        int height = image.getHeight();

        for (int y = 0; y < height; y++) {
            for (int x = 0; x < width; x++) {
                Color color = new Color(image.getRGB(x, y));
                // Calculate grayscale brightness
                int gray = (color.getRed() + color.getGreen() + color.getBlue()) / 3;

                // Map brightness to ASCII character index
                int index = (gray * (ASCII_CHARS.length() - 1)) / 255;
                ascii.append(ASCII_CHARS.charAt(index));
            }
            ascii.append("\n");
        }
        return ascii.toString();
    }

    public static void main(String[] args) {
        try {
            BufferedImage image = ImageIO.read(new File("input.jpg"));
            System.out.println(convertToAscii(image));
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
